		virtual bool ProcessCmdKey(System::Windows::Forms::Message% msg, System::Windows::Forms::Keys keyData) override
		{
			if (pnlMenu->Visible || pnlExtras->Visible)
				return Form::ProcessCmdKey(msg, keyData);

			Direction newDir = currentDir;
			bool isDirectionKey = false;
			
			if (!game_over) {
				if (keyData == Keys::W) { if (currentDir != Direction::Down) newDir = Direction::Up; isDirectionKey = true; }
				else if (keyData == Keys::S) { if (currentDir != Direction::Up) newDir = Direction::Down; isDirectionKey = true; }
				else if (keyData == Keys::A) { if (currentDir != Direction::Right) newDir = Direction::Left; isDirectionKey = true; }
				else if (keyData == Keys::D) { if (currentDir != Direction::Left) newDir = Direction::Right; isDirectionKey = true; }

			}
			
			if (isDirectionKey) {
				if (!isGameRunning) {
					isGameRunning = true;
					currentDir = newDir;
					gameTimer->Start();
				}
				else if (newDir != currentDir) nextDir = newDir;
				return true;
			}
			return Form::ProcessCmdKey(msg, keyData);
		}

		void InitializeGame() {
			mods = gcnew GameModifiers();
			rand = gcnew Random();
			snake = gcnew array<int, 2>(200, 2);
			ResetGame();
		}

		void ResetGame() {
			isGameRunning = false;
			score = 0;
			snakeLength = 3;
			currentDir = Direction::Stop;
			nextDir = Direction::Stop;

			
			SpeedTimer->Stop();
			portalTimer->Stop();
			SpeedBoostActice = false;
			portalsActive = false;

			
			snake[0, 0] = 3; snake[0, 1] = 5; 
			snake[1, 0] = 2; snake[1, 1] = 5;
			snake[2, 0] = 1; snake[2, 1] = 5;

			mods->portals = chkPortals->Checked;
			mods->speedFood = chkSpeedFood->Checked;
			mods->enhancedFood = chkEnhanced->Checked;
			mods->wrapMode = chkWrap->Checked;
			mods->christmas = chkchristmas->Checked;

			SpawnFood();
			SpawnPortals();
			
			lblGameOver->Visible = false;
		}

		void SpawnFood() {
			bool valid = false;
			while (!valid) {
				foodX = rand->Next(0, WIDTH);
				foodY = rand->Next(0, HEIGHT);
				valid = true;
				for (int i = 0; i < snakeLength; i++) {
					if (snake[i, 0] == foodX && snake[i, 1] == foodY) { valid = false; break; }
					if(portal1X == foodX && portal1Y == foodY || portal2X == foodX && portal2Y == foodY) { valid = false; break;}
				}
			}
			int chance = rand->Next(0, 100);               //chances { % }
			if (mods->speedFood && chance > 80) foodType = 2;
			else if (mods->enhancedFood && chance > 40) foodType = 1;
			else foodType = 0;
		}

		void SpawnPortals() {
			if (!mods->portals) return;

		
			if (portalsActive) return;

			// 30% Chance to Spawn
			if (rand->Next(0, 100) < 30) { //chances { % }
				bool v1 = false, v2 = false;
				while (!v1) {
					portal1X = rand->Next(0, WIDTH); portal1Y = rand->Next(0, HEIGHT);
					v1 = true;
					for (int i = 0; i < snakeLength; i++) if (snake[i, 0] == portal1X && snake[i, 1] == portal1Y) v1 = false;
					if (portal1X == foodX && portal1Y == foodY) v1 = false;
				}
				while (!v2) {
					portal2X = rand->Next(0, WIDTH); portal2Y = rand->Next(0, HEIGHT);
					v2 = true;
					if (portal2X == portal1X && portal2Y == portal1Y) v2 = false;
					for (int i = 0; i < snakeLength; i++) if (snake[i, 0] == portal2X && snake[i, 1] == portal2Y) v2 = false;
					if (portal2X == foodX && portal2Y == foodY) v2 = false;
				}

				
				portalsActive = true;
				portalTimer->Start();
			}
		}

		
		void OnPortalTimerTick(System::Object^ sender, System::EventArgs^ e) {
			
			if (snake[0, 0] == portal1X && snake[0, 1] == portal1Y) return;
			if (snake[0, 0] == portal2X && snake[0, 1] == portal2Y) return;

			portalsActive = false;
			portalTimer->Stop();
		}

		void SpeedBoostTimer(System::Object^ sender, System::EventArgs^ e) {
		
			SpeedTimer->Stop();
			SpeedBoostActice = false;
		}

		void GameTick(System::Object^ sender, System::EventArgs^ e) {
			if (nextDir != Direction::Stop) { currentDir = nextDir; nextDir = Direction::Stop; }
			if (currentDir == Direction::Stop) return;

			for (int i = snakeLength - 1; i > 0; i--) {
				snake[i, 0] = snake[i - 1, 0];
				snake[i, 1] = snake[i - 1, 1];
			}

			if (currentDir == Direction::Up) snake[0, 1]--;
			else if (currentDir == Direction::Down) snake[0, 1]++;
			else if (currentDir == Direction::Left) snake[0, 0]--;
			else if (currentDir == Direction::Right) snake[0, 0]++;

			if (mods->wrapMode) {
				if (snake[0, 0] < 0) snake[0, 0] = WIDTH - 1;
				if (snake[0, 0] >= WIDTH) snake[0, 0] = 0;
				if (snake[0, 1] < 0) snake[0, 1] = HEIGHT - 1;
				if (snake[0, 1] >= HEIGHT) snake[0, 1] = 0;
			}
			else if (snake[0, 0] < 0 || snake[0, 0] >= WIDTH || snake[0, 1] < 0 || snake[0, 1] >= HEIGHT) {
				GameOver(); return;
			}
			
			CheckCollisions();
			gameCanvas->Invalidate();
			
			lblScore->Text = "Score: " + score;
		}

		void CheckCollisions() {
			for (int i = 1; i < snakeLength; i++) {
				if (snake[i, 0] == snake[0, 0] && snake[i, 1] == snake[0, 1]) { GameOver(); return; }
			}
			if (snake[0, 0] == foodX && snake[0, 1] == foodY) {
				if (foodType == 0) { score += 5; snakeLength++; lastfoodeaten = 0; }
				else if (foodType == 1) { score += 10; snakeLength += 2; lastfoodeaten = 1; }
				else if (foodType == 2) {

					score += 5;
					snakeLength++;
					lastfoodeaten = 2;
					SpeedBoostActice = true;
					SpeedTimer->Start();
				}

				
				SpawnFood();
				SpawnPortals();
			}
			if (portalsActive) {
				if (snake[0, 0] == portal1X && snake[0, 1] == portal1Y) { snake[0, 0] = portal2X; snake[0, 1] = portal2Y; }
				else if (snake[0, 0] == portal2X && snake[0, 1] == portal2Y) { snake[0, 0] = portal1X; snake[0, 1] = portal1Y; }
			}
		}

		void GameOver() {
			game_over = true;
			gameTimer->Stop();
			portalTimer->Stop(); 
			SpeedTimer->Stop();
			isGameRunning = false;
			lblGameOver->Visible = true;
			lblGameOver->BringToFront();
			Timer^ t = gcnew Timer();
			t->Interval = 1000;
			t->Tick += gcnew EventHandler(this, &GUI::ReturnToMenu);
			t->Start();
		}

		void ReturnToMenu(Object^ sender, EventArgs^ e) {
			game_over = false;
			((Timer^)sender)->Stop();
			pnlMenu->Visible = true;
			lblGameOver->Visible = false;
		}

		GraphicsPath^ GetRoundedRect(Rectangle rect, int radius) {
			GraphicsPath^ path = gcnew GraphicsPath();
			int d = radius * 2;
			path->AddArc(rect.X, rect.Y, d, d, 180, 90);
			path->AddArc(rect.Right - d, rect.Y, d, d, 270, 90);
			path->AddArc(rect.Right - d, rect.Bottom - d, d, d, 0, 90);
			path->AddArc(rect.X, rect.Bottom - d, d, d, 90, 90);
			path->CloseFigure();
			return path;

		}

		void DrawSantaHat(Graphics^ g, int x, int y) {
			Point pt1 = Point(x + 5, y + 10);
			Point pt2 = Point(x + 35, y + 10);
			Point pt3 = Point(x + 20, y - 10);
			array<Point>^ hatPoints = { pt1, pt2, pt3 };
			g->FillPolygon(Brushes::Crimson, hatPoints);
			g->FillRectangle(Brushes::White, x + 5, y + 8, 30, 4);
			g->FillEllipse(Brushes::White, x + 16, y - 15, 10, 8);
		}

		void OnDraw(System::Object^ sender, System::Windows::Forms::PaintEventArgs^ e) {



			if (mods->christmas) {
				gameCanvas->BackColor = Color::FromArgb(5, 15, 30);
			}

			if (SpeedBoostActice) {
				gameTimer->Interval = 150;
			}
			else
			{
				gameTimer->Interval = 400;
			}


			Graphics^ g = e->Graphics;
			g->SmoothingMode = SmoothingMode::AntiAlias;


			/*Pen^ gp = gcnew Pen(Color::FromArgb(30, 30, 30));
			for (int x = 0; x <= WIDTH; x++) g->DrawLine(gp, x * CELL_SIZE, 0, x * CELL_SIZE, HEIGHT * CELL_SIZE);
			for (int y = 0; y <= HEIGHT; y++) g->DrawLine(gp, 0, y * CELL_SIZE, WIDTH * CELL_SIZE, y * CELL_SIZE);*/

			

			// --- 3D APPLE RENDERING ---
			int fx = foodX * CELL_SIZE; int fy = foodY * CELL_SIZE;

			if (!mods->christmas)
			{
				g->DrawCurve(gcnew Pen(Color::SaddleBrown, 3), gcnew array<Point>{Point(fx + 20, fy + 12), Point(fx + 22, fy + 2)});
				g->FillEllipse(Brushes::ForestGreen, fx + 22, fy + 4, 10, 5);
				Color pCol = (foodType == 1) ? Color::Gold : (foodType == 2 ? Color::Cyan : Color::Crimson);
				Color dCol = (foodType == 1) ? Color::DarkGoldenrod : (foodType == 2 ? Color::DarkCyan : Color::DarkRed);
				Rectangle fr = Rectangle(fx + 6, fy + 10, CELL_SIZE - 12, CELL_SIZE - 14);
				LinearGradientBrush^ ab = gcnew LinearGradientBrush(fr, pCol, dCol, 90.0f);
				g->FillEllipse(ab, fr);
				g->FillEllipse(gcnew SolidBrush(Color::FromArgb(180, Color::White)), fx + 12, fy + 14, 8, 5);
			}
			else
			{
				if (foodType == 0) { // Snowball
					Rectangle snowRect = Rectangle(fx + 6, fy + 6, 30, 28);

					Color topColor = Color::White;
					Color bottomColor = Color::FromArgb(90, 105, 105);

					LinearGradientBrush^ snowBrush = gcnew LinearGradientBrush(snowRect, topColor, bottomColor, 90.0f);

					g->FillEllipse(snowBrush, snowRect);
					delete snowBrush;
				}
				else { // Gift
					Color giftCol = (foodType == 2) ? Color::RoyalBlue : Color::Gold;
					Color ribCol = (foodType == 2) ? Color::White : Color::Crimson;
					g->FillRectangle(gcnew SolidBrush(giftCol), fx + 8, fy + 10, 24, 24);
					g->DrawLine(gcnew Pen(ribCol, 5), fx + 18, fy + 10, fx + 18, fy + 34);
					g->DrawLine(gcnew Pen(ribCol, 4), fx + 8, fy + 20, fx + 32, fy + 20);
				}

			}

			if (portalsActive) {
				g->DrawEllipse(gcnew Pen(Color::MediumPurple, 4), portal1X * CELL_SIZE + 7, portal1Y * CELL_SIZE + 4, CELL_SIZE - 14, CELL_SIZE - 8);
				g->DrawEllipse(gcnew Pen(Color::Orange, 4), portal2X * CELL_SIZE + 7, portal2Y * CELL_SIZE + 4, CELL_SIZE - 14, CELL_SIZE - 8);
			}

			// --- SMOOTH SNAKE RENDERING ---
			for (int i = 0; i < snakeLength; i++) {
				int sx = snake[i, 0] * CELL_SIZE; int sy = snake[i, 1] * CELL_SIZE;
				Rectangle r = Rectangle(sx + 2, sy + 2, CELL_SIZE - 4, CELL_SIZE - 4);
				if (i == 0) {

					if (SpeedBoostActice) {
					
						LinearGradientBrush^ hb = gcnew LinearGradientBrush(r, Color::LightBlue, Color::Blue, 45.0f);
						g->FillPath(hb, GetRoundedRect(r, 12));
					}
					else
					{
						
						LinearGradientBrush^ hb = gcnew LinearGradientBrush(r, Color::Lime, Color::ForestGreen, 45.0f);
						g->FillPath(hb, GetRoundedRect(r, 12));
					}

					//eye
					if (lastfoodeaten == 1) {
						g->FillEllipse(Brushes::Gold, sx + 10, sy + 10, 10, 10);
						g->FillEllipse(Brushes::Gold, sx + 22, sy + 10, 10, 10);
					}
					else
					{
						g->FillEllipse(Brushes::White, sx + 10, sy + 10, 10, 10);
						g->FillEllipse(Brushes::White, sx + 22, sy + 10, 10, 10);
					}
					g->FillEllipse(Brushes::Black, sx + 13, sy + 13, 3, 3);
					g->FillEllipse(Brushes::Black, sx + 25, sy + 13, 3, 3);

				}
				else {
					
					int taper = (i > 15) ? 10 : (i / 2);
					Rectangle br = Rectangle(sx + 2 + taper, sy + 2 + taper, CELL_SIZE - 4 - (taper * 2), CELL_SIZE - 4 - (taper * 2));
					int gVal = 225 - (i * 5 > 150 ? 150 : i * 5);

					if (SpeedBoostActice) {
				
						g->FillPath(gcnew SolidBrush(Color::FromArgb(100, 158, gVal)), GetRoundedRect(br, 8));
					}
					else
					{
						
						g->FillPath(gcnew SolidBrush(Color::FromArgb(0, gVal, 0)), GetRoundedRect(br, 8));
					}

					if (mods->christmas) {

						int hatx = snake[0, 0] * CELL_SIZE;
						int haty = snake[0, 1] * CELL_SIZE;
						DrawSantaHat(g, hatx, haty);
					}
				}
			}
		}
